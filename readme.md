# Via Release
<code>
tar -zxvf cla.tar.gz
cd .cla
./conf <ident> <port>
./cla
./start
</code>

<code>
git clone https://github.com/duniairc/cla.git
cd cla 
chmod +x cla camelia start claa
./cla # for add contrab
./start
</code>
